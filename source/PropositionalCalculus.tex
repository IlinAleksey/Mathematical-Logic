\begin{lemma}
Если $\phi \fCenter \psi$ выводима, то из $\Gamma \fCenter \phi$ можно вывести $\Gamma \fCenter \psi$
\begin{proof} 

\begin{prooftree}
		\AxiomC{$\phi \fCenter \psi$}
	\UnaryInfC{$\fCenter \phi \to \psi$}
	\UnaryInfC{$...$}
\UnaryInfC{$\Gamma\fCenter \phi \to \psi$}
\AxiomC{$\Gamma \fCenter \phi$}
\BinaryInfC{$\Gamma \fCenter \psi$}
\end{prooftree}

\end{proof}
\end{lemma}


\begin{lemma}
Если $\psi \fCenter \phi$ выводима, то из $\Gamma,\phi \fCenter \theta$ можно вывести $\Gamma,\psi \fCenter \theta$
\begin{proof}

\begin{prooftree}
		\AxiomC{$\psi \fCenter \phi$}
		\UnaryInfC{$...$}
	\UnaryInfC{$\Gamma, \psi \fCenter \phi$}
	
\AxiomC{$\Gamma,\phi \fCenter \theta$}
\UnaryInfC{$\Gamma \fCenter \phi \to \theta$}
\UnaryInfC{$\Gamma, \psi \fCenter \phi \to \theta$}
\BinaryInfC{$\Gamma,\psi \fCenter \theta$}
\end{prooftree}

\end{proof}
\end{lemma}
